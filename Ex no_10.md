#include <stdio.h>

int main() {
    int n, i;
    printf("Enter the number of times to print 'KEYBOARD': ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("KEYBOARD\n");
    }

    return 0;
}
